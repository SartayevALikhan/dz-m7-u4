# 💡 Решение
# GreenEye — это программное решение на базе машинного обучения, которое:

# Анализирует открытые спутниковые снимки (Sentinel-2).

# Детектирует изменения в плотности растительности с точностью до 95%.

# Автоматически отправляет алерты экологическим службам и публикует данные в открытый реестр.

# 🚀 Как это помогает
# Скорость: Обнаружение вырубки за 24 часа вместо недель.

# Прозрачность: Данные невозможно скрыть — они доступны общественности.

# Масштаб: Одна модель может следить за всей Амазонией или Тайгой одновременно.

import numpy as np

def calculate_ndvi(nir_band, red_band):
    """
    Рассчитывает индекс вегетации. 
    Значения близкие к 1.0 — густой лес, к 0.0 — голая земля.
    """
    # Избегаем деления на ноль
    ndvi = (nir_band - red_band) / (nir_band + red_band + 1e-10)
    return ndvi

def detect_deforestation(ndvi_before, ndvi_after, threshold=0.2):
    """
    Сравнивает два снимка и находит зоны вырубки.
    """
    diff = ndvi_before - ndvi_after
    # Если разница больше порога — фиксируем аномалию
    alerts = diff > threshold
    return alerts

# Пример данных (условно)
nir_t1 = np.array([0.8, 0.7, 0.9]) # Снимок "До"
red_t1 = np.array([0.1, 0.1, 0.1])
ndvi_past = calculate_ndvi(nir_t1, red_t1)

print(f"Текущий индекс здоровья леса: {ndvi_past}")
